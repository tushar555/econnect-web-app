  checkVisibility(list) {
    if (list.index === 5) {
      if (this.IsITEnabled) {
        return true;
      } else {
        return false;
      }
    } else if (list.index === 17) {
      if (this.IsManager) {
        return true;
      } else {
        return false;
      }
    } else if (list.index === 16) {
      if (this.IsTimeAdmin) {
        return true;
      } else {
        return false;
      }
    } else if (list.index === 15) {
      if (this.IsCSR) {
        return true;
      } else {
        return false;
      }
    } else if (list.index === 18) {
      if (
        this.TokenId === "23175693" ||
        this.TokenId === "23135063" ||
        this.TokenId === "24009108" ||
        this.IsRHR
      ) {
        return true;
      } else {
        return false;
      }
    } else if (list.index === 20) {
      if (this.IsGiftEligible) {
        return true;
      } else {
        return false;
      }
    } else if (list.index === 21) {
      if (
        this.companyName === "MIBL" ||
        this.companyName === "MMFSL" ||
        this.companyName === "MAMC"
      ) {
        return true;
      } else {
        return false;
      }
    } else {
      return true;
    }
  }
